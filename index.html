/** * 구글 스프레드시트의 A열에 값이 입력될 때 작동하는 트리거 함수입니다. * 1-10: 노란색 * 11-20: 초록색 * 21-30: 빨간색 * 그 외의 값 또는 빈 셀: 배경색 초기화 (없음) * * @param {Object} e 편집 이벤트 객체 */ function onEdit(e) { // 이벤트 객체가 없거나 편집된 범위가 없으면 함수를 종료합니다. if (!e || !e.range) { return; } const range = e.range; const sheet = range.getSheet(); const column = range.getColumn(); // A열(1번째 열)이 아닌 경우 작동하지 않도록 제한합니다. if (column !== 1) { return; } const row = range.getRow(); // 첫 번째 행이 헤더(제목)일 수 있으므로, 2번째 행부터 적용하려면 아래 조건의 주석을 해제하세요. // if (row < 2) return; const value = range.getValue(); // 입력된 값이 숫자인지 확인하고 범위에 따라 색상을 지정합니다. // 값이 비어있지 않고, 숫자형태인 경우에만 처리합니다. if (value !== "" && !isNaN(value)) { const numValue = Number(value); if (numValue >= 1 && numValue <= 10) { // 1 ~ 10: 노란색 (#ffe599) range.setBackground("#ffe599"); } else if (numValue >= 11 && numValue <= 20) { // 11 ~ 20: 초록색 (#b6d7a8) range.setBackground("#b6d7a8"); } else if (numValue >= 21 && numValue <= 30) { // 21 ~ 30: 빨간색 (#ea9999) range.setBackground("#ea9999"); } else { // 지정 범위를 벗어난 숫자는 배경색을 초기화합니다. range.setBackground(null); } } else { // 빈 셀이 되거나 숫자가 아닌 값이 입력되면 배경색을 초기화합니다. range.setBackground(null); } }
